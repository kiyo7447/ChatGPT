import shutil

year = input('年：')
month = input('月：')
client = 'C運輸'

if int(month) <= 6:
    half = '(上半期)'
else:
    half = '(下半期)'

fname = '請求書' + client + '様' + year \
    + '年' + month + '月' + half + '.xlsx'

path_src = 'template\\請求書ひな形.xlsx'
shutil.copy(path_src, 'invoice\\' + fname)
